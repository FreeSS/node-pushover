node-pushover.

node-pushover exposes a simple to use API for pushover.net's notification system.

$ npm install pushover-notifications

Usage

var push = require( 'pushover-notifications' );

var p = new push( {
    user: '',
    token: ''
});

var msg = {
    message: 'omg node test',
    title: "Well - this is fantastic",
    //priority: 1,
};

p.send( msg, function( err, result ) {
    if ( err ) {
        throw err;
    }

    console.log( result );
});
